334 /* ****************************************************************
335  Class to implement an event
336 
337  There are four components to an event: start time, stop time
338  nest_level, and timer_name.
339  The start and stop time should be obvious (recorded in clock ticks).
340  The nest_level relates to the bar width in the timeline graph.
341  The timer_name is used to determine which timer event triggered this event.
342 
343  the interface to this class is through four read-only operations:
344  1) getStart() -- returns the start time as 64 bit integer
345  2) getStop() -- returns the stop time as 64 bit integer
346  3) getNestLevel() -- returns the nest level of the event
347  4) getTimerName() -- returns the timer name that triggered event
348 
349  *MORE ON NEST_LEVEL*
350  The nest level is used in the bar graph that represents the timeline.
351  Its main purpose is for showing how events are nested inside eachother.
352  For example, say events, A, B, and C are recorded. If the timeline
353  looks like this:
354 
355 Begin -------------------------------------------------------------> Time
356  | | | | | |
357  A B C C B A
358  start start start end end end
359 
360  Then A, B, C will have a nest level of 1, 2, 3 respectively.
361  These values are then used to calculate the barwidth so you can
362  see that inside A, B has occurred, and inside B, C has occurred.
363  Currently, this is shown with A's bar width being larger than B's
364  bar width, and B's bar width being larger than C's bar width.
365 
366 **************************************************************** */
367 class kmp_stats_event {
368  uint64_t start;
369  uint64_t stop;
370  int nest_level;
371  timer_e timer_name;
372  public:
373  kmp_stats_event() : start(0), stop(0), nest_level(0), timer_name(TIMER_LAST) {}
374  kmp_stats_event(uint64_t strt, uint64_t stp, int nst, timer_e nme) : start(strt), stop(stp), nest_level(nst), timer_name(nme) {}
375  inline uint64_t getStart() const { return start; }
376  inline uint64_t getStop() const { return stop; }
377  inline int getNestLevel() const { return nest_level; }
378  inline timer_e getTimerName() const { return timer_name; }

381 /* ****************************************************************
382  Class to implement a dynamically expandable array of events
383 
384  ---------------------------------------------------------
385  | event 1 | event 2 | event 3 | event 4 | ... | event N |
386  ---------------------------------------------------------
387 
388  An event is pushed onto the back of this array at every
389  explicitTimer->stop() call. The event records the thread #,
390  start time, stop time, and nest level related to the bar width.
391 
392  The event vector starts at size INIT_SIZE and grows (doubles in size)
393  if needed. An implication of this behavior is that log(N)
394  reallocations are needed (where N is number of events). If you want
395  to avoid reallocations, then set INIT_SIZE to a large value.
396 
397  the interface to this class is through six operations:
398  1) reset() -- sets the internal_size back to 0 but does not deallocate any memory
399  2) size() -- returns the number of valid elements in the vector
400  3) push_back(start, stop, nest, timer_name) -- pushes an event onto
401  the back of the array
402  4) deallocate() -- frees all memory associated with the vector
403  5) sort() -- sorts the vector by start time
404  6) operator[index] or at(index) -- returns event reference at that index
405 
406 **************************************************************** */
407 class kmp_stats_event_vector {
408  kmp_stats_event* events;
409  int internal_size;
410  int allocated_size;
411  static const int INIT_SIZE = 1024;
412  public:
413  kmp_stats_event_vector() {
414  events = (kmp_stats_event*)__kmp_allocate(sizeof(kmp_stats_event)*INIT_SIZE);
415  internal_size = 0;
416  allocated_size = INIT_SIZE;
417  }
418  ~kmp_stats_event_vector() {}
419  inline void reset() { internal_size = 0; }
420  inline int size() const { return internal_size; }
421  void push_back(uint64_t start_time, uint64_t stop_time, int nest_level, timer_e name) {
422  int i;
423  if(internal_size == allocated_size) {
424  kmp_stats_event* tmp = (kmp_stats_event*)__kmp_allocate(sizeof(kmp_stats_event)*allocated_size*2);
425  for(i=0;i<internal_size;i++) tmp[i] = events[i];
426  __kmp_free(events);
427  events = tmp;
428  allocated_size*=2;
429  }
430  events[internal_size] = kmp_stats_event(start_time, stop_time, nest_level, name);
431  internal_size++;
432  return;
433  }
434  void deallocate();
435  void sort();
436  const kmp_stats_event & operator[](int index) const { return events[index]; }
437  kmp_stats_event & operator[](int index) { return events[index]; }
438  const kmp_stats_event & at(int index) const { return events[index]; }
439  kmp_stats_event & at(int index) { return events[index]; }
440 };

444 
445  |---| ---> |---| ---> |---| ---> |---| ---> ... next
446  | | | | | | | |
447  |---| <--- |---| <--- |---| <--- |---| <--- ... prev
448  Sentinel first second third
449  Node node node node
450 
451  The Sentinel Node is the user handle on the list.
452  The first node corresponds to thread 0's statistics.
453  The second node corresponds to thread 1's statistics and so on...
454

456  hold that thread's statistics. The _explicitTimers

618  *
619  * @param name timer name as specified under the KMP_FOREACH_TIMER() macro

621  * \details Use KMP_TIME_BLOCK(name) macro to time a code block. This will record the time taken in the block
622  * and use the destructor to stop the timer. Convenient!
623  * With this definition you can't have more than one KMP_TIME_BLOCK in the same code block.
